1function OUTPUT_SCHEDULE = FORCE_DIRECTED_dot_cpp(operations, Latency)
  2
  3in = 0;        % input nop
  4out = inf;     % output nop
  5
  6if nargin == 0
  7    % Test Case and input discription
  8    % Values got from the FDS example in the HLSM_Scheduling slides
  9    
 10    % number each operation node
 11    % order doesn't matter as long as it's consistant
 12    operations.numbers = 1:11;
 13    
 14    % a record of all of the 'children' of each operation
 15    % a 'child' is a an operation that will depend on the current operation
 16    % operations.children{5} will be an array of all of the 'children' for
 17    % the operation that was given the node number '5'
 18    operations.children = {[3] ...
 19        [3] ...
 20        [4] ...
 21        [5] ...
 22        [out] ...
 23        [7] ...
 24        [5] ...
 25        [9] ...
 26        [out] ...
 27        [11] ...
 28        [out] ...
 29        };
 30    
 31    % a record of all of the 'parents' of each operation
 32    % a 'parent' is a an operation that the current operation depends on
 33    % operations.parents{5} will be an array of all of the 'parents' for
 34    % the operation that was given the node number '5'
 35    operations.parents = {[in] ...
 36        [in] ...
 37        [1, 2] ...
 38        [3] ...
 39        [4, 7] ...
 40        [in] ...
 41        [6] ...
 42        [in] ...
 43        [8] ...
 44        [in] ...
 45        [10] ...
 46        };
 47    
 48    % a record of which resource each node is using
 49    % Let's say the resource type is an enumuration with ALU = 1 and
 50    % MULT = 2, for this test case
 51    operations.resource = [ 2 ...
 52        2 ...
 53        2 ...
 54        1 ...
 55        1 ...
 56        2 ...
 57        2 ...
 58        2 ...
 59        1 ...
 60        1 ...
 61        1 ...
 62        ];
 63    
 64    % a record of which operations have been scheduled already
 65    % should always start out with everything as false
 66    operations.scheduled = [ false ...
 67        false ...
 68        false ...
 69        false ...
 70        false ...
 71        false ...
 72        false ...
 73        false ...
 74        false ...
 75        false ...
 76        false ...
 77        ];
 78    
 79    % Latency
 80    Latency = 6;
 81    % Latency was 4 in the in-class example. but that doesn't work when
 82    % MULT's have a latency of 2
 83    
 84end
 85
 86all_operations_scheduled = false;
 87
 88%init output
 89% the output of this function is a cell array where each cell represents
 90% a time cycle, and the contents of the cell is an array of all the
 91% nodes that should be scheduled in that cycle
 92% OUTPUT_SCHEDULE = cell(Latency,1);
 93OUTPUT_SCHEDULE = zeros(numel(operations.numbers));
 94
 95
 96% ALU is resource 1 with latency of 1 cycle
 97% MULT is resource 2 with latency of 2 cycles
 98% DIV/MOD is resource 3 with latency of 3 cycles
 99Resource_Latency = [ 1 ...
100    2 ...
101    3 ...
102    ];
103
104
105%% Compute the time frames
106Time_Frames = zeros(numel(operations.numbers), 2);
107
108%ASAP Schedule
109
110% ASAP = cell(Latency,1);
111ASAP = -1*ones(Latency,numel(operations.numbers));
112parent_count = zeros(numel(operations.numbers),1);
113op_num = 1;
114
115% loop through each node
116for v = 1:numel(operations.numbers)
117    % find how many generations of parents this node has
118    parent_count(operations.numbers(v)) = count_parents(operations.numbers(v), parent_count(operations.numbers(v)), operations, Resource_Latency);
119
120    if parent_count(operations.numbers(v)) < Latency
121        % the number of generations of parents is the location in the ASAP Schedule
122        op_num = numel(operations.numbers) - sum(ASAP(parent_count(operations.numbers(v))+1,:) > -1) + 1;
123        ASAP(parent_count(operations.numbers(v))+1,op_num) = operations.numbers(v);
124        Time_Frames(operations.numbers(v),1) = parent_count(operations.numbers(v))+1; % First part of Time Frame is Soonest possible time
125    else
126        %disp(['Cant Schedule Node #' num2str(v) ' in ASAP'])
127        disp(['Cant Schedule This Node in ASAP'])
128    end
129end
130
131
132
133%ALAP Schedule
134
135% ALAP = cell(Latency,1);
136ALAP = -1*ones(Latency,numel(operations.numbers))
137child_count = zeros(numel(operations.numbers),1);
138op_num = 1;
139
140% Look for operations that have no children
141for v = 1:numel(operations.numbers)
142    % find how many generations of children this node has
143    child_count(operations.numbers(v)) = count_children(operations.numbers(v), child_count(operations.numbers(v)), operations, Resource_Latency);
144    
145    if child_count(operations.numbers(v)) < Latency
146        % the number of generations of children is the reverse location in the ALAP Schedule
147        op_num = numel(operations.numbers) - sum(ALAP(Latency - child_count(operations.numbers(v)),:) > -1) + 1;
148        ALAP(Latency - child_count(operations.numbers(v)),op_num) = operations.numbers(v);
149        Time_Frames(operations.numbers(v),2) = Latency - child_count(operations.numbers(v)); % Second part of Time Frame is Latest possible time
150    else
151        %disp(['Cant Schedule Node #' num2str(v) ' in ALAP'])
152        disp(['Cant Schedule This Node in ASAP'])
153    end
154end
155
156
157while ~all_operations_scheduled
158    
159    %% Compute the operations and type probabilities
160    Operation_Probability = zeros(numel(operations.numbers), Latency);
161    Type_Probability = zeros(numel(unique(operations.resource)), Latency);
162    
163    for t = 1:Latency      %Loop through each cycle
164        for v = 1:numel(operations.numbers)  %Loop through each operations
165            % if the current cycle is within the time frame for this operation
166            if t >= Time_Frames(v,1) && t <= Time_Frames(v,2)
167                % then the probability is the recipricol of the range of the Time Frame
168                Operation_Probability(v,t) = 1/(Time_Frames(v,2) - Time_Frames(v,1) + 1);
169            else
170                % else the probablility is 0
171                Operation_Probability(v,t) = 0;
172            end
173            Type_Probability(operations.resource(v),t) = Operation_Probability(v,t) + Type_Probability(operations.resource(v),t); % +=
174        end
175    end
176    
177    
178    %% Compute the self-forces, predecessor/succesor forces and total forces
179    % Calculate All Self Forces
180    Self_Force = zeros(numel(operations.numbers), Latency);
181    
182    for t = 1:Latency      %Loop through each cycle
183        for v = 1:numel(operations.numbers)  %Loop through each operations
184            % if the current cycle is within the time frame for this operation
185            if t >= Time_Frames(v,1) && t <= Time_Frames(v,2)
186                for c = 1:Latency %Loop through each cycle again
187                    Self_Force(v, t) = Type_Probability(operations.resource(v),c)*((c == t) - Operation_Probability(v,c)) + Self_Force(v, t); % +=
188                end
189            else
190                % set self force to Inf becuase operation can't be scheduled this cycle
191                Self_Force(v, t) = inf;
192            end
193        end
194    end
195    
196    % Calculate All Predecessor/Successor Forces
197    Predecessor_Force = zeros(numel(operations.numbers), Latency);
198    Successor_Force = zeros(numel(operations.numbers), Latency);
199    
200    for t = 1:Latency           %Loop through each cycle
201        for v = 1:numel(operations.numbers)     %Loop through each operations
202            % if the current cycle is within the time frame for this operation
203            if t >= Time_Frames(v,1) && t <= Time_Frames(v,2)
204                for p = 1:numel(operations.parents(v,:))  %Loop through all of this operation's parents
205                    if p>0
206                        % will scheduling v at time t effect the schedule for p?
207                        if t >= Time_Frames(p,1) && t <= Time_Frames(p,2)
208                            
209                            % Find all the options for scheduling the parent
210                            time_options = Time_Frames(p,1):Time_Frames(p,2);   % all times when p can be scheduled
211                            time_options = time_options(time_options ~= t);     % except the current time t
212                            
213                            %Loop through all these options to add up the force
214                            for o = 1:numel(time_options)
215                                % add the self force of this parent at this t
216                                Predecessor_Force(v, t) = Self_Force(p, time_options(o)) +  Predecessor_Force(v, t); % +=
217                            end
218                            
219                        else
220                            % Parent uneffected
221                            Predecessor_Force(v, t) = 0 + Predecessor_Force(v, t);  % temporary for bookkeeping
222                        end
223                    else
224                        Predecessor_Force(v, t) = 0 + Predecessor_Force(v, t);  % temporary for bookkeeping
225                    end
226                end
227                
228                for c = 1:numel(operations.children(v,:))  %Loop through all of this operation's children
229                    if c<inf
230                        % will scheduling v at time t effect the schedule for p?
231                        if t >= Time_Frames(c,1) && t <= Time_Frames(c,2)
232                            
233                            % Find all the options for scheduling the child
234                            time_options = Time_Frames(c,1):Time_Frames(c,2);   % all times when p can be scheduled
235                            time_options = time_options(time_options ~= t);     % except the current time t
236                            
237                            %Loop through all these options to add up the force
238                            for o = 1:numel(time_options)
239                                % add the self force of this child at this t
240                                Successor_Force(v, t) = Self_Force(c, time_options(o)) +  Successor_Force(v, t); % +=
241                            end
242                            
243                        else
244                            % Parent uneffected
245                            Successor_Force(v, t) = 0 + Successor_Force(v, t);  % temporary for bookkeeping
246                        end
247                    else
248                        Successor_Force(v, t) = 0 + Successor_Force(v, t);  % temporary for bookkeeping
249                    end
250                end
251                
252            else
253                Predecessor_Force(v, t) = 0;
254                Successor_Force(v, t) = 0;
255            end
256        end
257    end
258    
259    Total_Force = Self_Force + Predecessor_Force + Successor_Force;
260    
261    %I would sometimes get really small floating point umbers for Total
262    %Force, so that would screw up the min force calculation
263    %Total_Force = round(Total_Force, 5); % this rounds Total_Force to 5 decimal places, may not be nessesary
264    
265    %% Schedule the operations with the least force and update its time-frame
266    
267    [min_force_operation, min_force_cycle] = find(Total_Force == min(min(Total_Force)));    % need to find alternative to the 'find' function that will work with c++
268    
269    % if there are multple min Total Force, just pick the first one found that wasn't already scheduled
270    for o = 1:numel(min_force_operation)
271        if ~operations.scheduled(min_force_operation(o))
272            min_force_operation =  min_force_operation(o);
273            min_force_cycle = min_force_cycle(o);
274            break;
275        end
276    end
277    
278    % add the minimum Force node to the Output Schedule
279%     OUTPUT_SCHEDULE{min_force_cycle} = [min_force_operation, OUTPUT_SCHEDULE{min_force_cycle}];
280    out_index = 1;
281    out_index = find(operations.numbers == min_force_operation);
282    OUTPUT_SCHEDULE(out_index) = min_force_cycle;
283    
284    min_force_operation_int = 0;
285    min_force_operation_int = min_force_operation(1);
286    
287    min_force_cycle_int = 0;
288    min_force_cycle_int = min_force_cycle(1);
289    
290    % mark it as scheduled
291    operations.scheduled(min_force_operation_int) = true;
292    
293    % adjust the Time_Frames to lock that node in place
294    Time_Frames(min_force_operation_int,1) = min_force_cycle_int;
295    Time_Frames(min_force_operation_int,2) = min_force_cycle_int;
296    
297    % update the depedant Time_Frames (parents/children)
298    update_parent_Time(min_force_operation_int, min_force_cycle_int, operations, Time_Frames);
299    update_child_Time(min_force_operation_int, min_force_cycle_int, operations, Time_Frames);
300    
301    
302    all_operations_scheduled = all(operations.scheduled);
303    
304end
305
306end

308%% Other Functions
309function parent_count = count_parents(v, parent_count, operations, Resource_Latency)
310local_family = operations.parents(v,:);
311if sum(local_family) == 0   %only has "in" as parent
312    return;
313else
314    local_parent_count = zeros(numel(local_family),1) + parent_count;
315    for f = 1:numel(local_family)
316        local_parent_count(local_family(f)) = count_parents(local_family(f), local_parent_count(local_family(f)) + ...
317            1*Resource_Latency(operations.resource(v)),  operations, Resource_Latency );
318    end
319end
320parent_count = max(local_parent_count);
321end
322

323function child_count = count_children(v, child_count, operations, Resource_Latency)
324local_family = operations.children(v,:);
325if all(local_family == inf)   %only has "out" as child
326    return;
327else
328    local_child_count = zeros(numel(local_family),1) + child_count;
329    for f = 1:numel(local_family)
330        local_child_count(local_family(f)) = count_children(local_family(f), local_child_count(local_family(f)) + ...
331            1*Resource_Latency(operations.resource(v)),  operations, Resource_Latency);
332    end
333end
334child_count = max(local_child_count);
335end
336
337function update_parent_Time(v, t, operations, Time_Frames)
338local_family = zeros(numel(operations.parents(v,:)),1);
339local_family = operations.parents(v,:);
340for f = 1:numel(local_family)
341    if local_family(f)>0
342        % if this parent was effected by the sceduled operations
343        if t >= Time_Frames(local_family(f),1) && t <= Time_Frames(local_family(f),2)
344            % then the latest possible time has changed for the parent
345            Time_Frames(local_family(f),2) = Time_Frames(v,1)-1;
346            update_parent_Time(local_family(f), Time_Frames(local_family(f),2), operations, Time_Frames);
347        end
348    end
349end
350end
351
352function update_child_Time(v, t, operations, Time_Frames)
353local_family = operations.children(v,:);
354for f = 1:numel(local_family)
355    if local_family(f)<inf
356        % if this child was effected by the sceduled operations
357        if t >= Time_Frames(local_family(f),1) && t <= Time_Frames(local_family(f),2)
358            % then the latest possible time has changed for the child
359            Time_Frames(local_family(f),1) = Time_Frames(v,2)+1;
360            update_child_Time(local_family(f), Time_Frames(local_family(f),1), operations, Time_Frames);
361        end
362    end
363end
